```compact
pragma language_version 0.21;

export ledger result: Uint<16>;

export circuit addTwoNums(num1: Uint<16>, num2: Uint<16>): [] {
    assert((num1 > 0 && num2 > 0), "no negative numbers allowed");
    result = disclose(num1 + num2) as Uint<16>;
}

export circuit subtractNums(num1: Uint<16>, num2: Uint<16>): [] {
    assert(((num1 > 0 && num2 > 0) && num1 > num2), "no operations resulting in negative numbers allowed");
    result = disclose(num1 - num2) as Uint<16>;
}

export circuit multiplyNums(num1: Uint<16>, num2: Uint<16>): [] {
    result = disclose(num1 * num2) as Uint<16>;
}

export circuit squareNum(num1: Uint<16>): [] {
    result = disclose(num1 * num1) as Uint<16>;
}
```
