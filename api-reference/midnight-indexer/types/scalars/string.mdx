import { useState } from "react";

export const Details = ({
  dataOpen,
  dataClose,
  children,
  startOpen = false
}) => {
  const [open, setOpen] = useState(startOpen);
  return (
    <details
      {...(open ? { open: true } : {})}
      className="details"
      style={{
        border: "none",
        boxShadow: "none",
        background: "var(--ifm-background-color)"
      }}
    >
      <summary
        onClick={(e) => {
          e.preventDefault();
          setOpen((open) => !open);
        }}
        style={{ listStyle: "none" }}
      >
        {open ? dataOpen : dataClose}
      </summary>
      {open && children}
    </details>
  );
};
